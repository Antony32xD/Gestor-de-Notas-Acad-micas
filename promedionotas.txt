INICIO

    // ==================== DECLARACIÓN DE LISTAS Y ESTRUCTURAS ====================
    cursos_lista ← ["Mate Discreta", "Contabilidad 2", "Pre Cálculo", "Algebra Lineal", "Algoritmos"]
    notas_lista ← [85, 65, 61, 84, 99]
    cola_revisiones ← Cola vacía
    historial_cambios ← Lista vacía

    // ==================== FUNCIONES ====================
    
    FUNCION limpiar_terminal
        SI sistema es Windows ENTONCES
            ejecutar "cls"
        SINO
            ejecutar "clear"
        FIN SI
    FIN FUNCION

    FUNCION mostrar_menu
        IMPRIMIR menú con opciones 1 a 13
    FIN FUNCION

    FUNCION registrar_curso_lista
        LEER cantidad n de cursos
        limpiar_terminal()
        PARA i ← 1 HASTA n HACER
            LEER nombre curso
            VALIDAR que no esté vacío ni repetido
            LEER nota curso
            VALIDAR que esté entre 0 y 100
            AGREGAR curso a cursos_lista
            AGREGAR nota a notas_lista
            limpiar_terminal()
        FIN PARA
        IMPRIMIR cursos y notas registrados
    FIN FUNCION

    FUNCION mostrar_cursos_lista
        SI cursos_lista vacía ENTONCES
            IMPRIMIR "No hay cursos"
        SINO
            IMPRIMIR todos los cursos y notas
        FIN SI
    FIN FUNCION

    FUNCION promedio
        SI cursos_lista vacía ENTONCES
            IMPRIMIR "No hay cursos"
        SINO
            CALCULAR promedio de notas_lista
            IMPRIMIR promedio
        FIN SI
    FIN FUNCION

    FUNCION cursos_aprobados_reprobados
        SI cursos_lista vacía ENTONCES
            IMPRIMIR "No hay cursos"
        SINO
            aprobados ← []
            reprobados ← []
            PARA cada curso en cursos_lista HACER
                SI nota ≥ 61 ENTONCES
                    AGREGAR a aprobados
                SINO
                    AGREGAR a reprobados
                FIN SI
            FIN PARA
            IMPRIMIR aprobados y reprobados
        FIN SI
    FIN FUNCION

    FUNCION buscar_curso
        LEER nombre curso
        SI nombre está en cursos_lista ENTONCES
            IMPRIMIR nota y estado (Aprobado/Reprobado)
        SINO
            IMPRIMIR "Curso no registrado"
        FIN SI
    FIN FUNCION

    FUNCION actualizar_nota
        LEER nombre curso
        SI curso existe EN cursos_lista ENTONCES
            LEER nueva nota
            VALIDAR nota entre 0 y 100
            GUARDAR cambio en historial_cambios
            ACTUALIZAR nota en notas_lista
        SINO
            IMPRIMIR "Curso no registrado"
        FIN SI
    FIN FUNCION

    FUNCION borrar_curso_lista
        MOSTRAR lista de cursos
        LEER número de curso a borrar
        CONFIRMAR eliminación
        ELIMINAR curso y nota de las listas
        limpiar_terminal()
        IMPRIMIR confirmación
    FIN FUNCION

    FUNCION ordenar_por_nota_burbuja
        ORDENAR cursos y notas de menor a mayor según nota
        IMPRIMIR lista ordenada
    FIN FUNCION

    FUNCION ordenar_por_nombre
        ORDENAR cursos y notas alfabéticamente
        IMPRIMIR lista ordenada
    FIN FUNCION

    FUNCION buscar_curso_binario
        LEER nombre curso
        ORDENAR cursos alfabéticamente
        REALIZAR búsqueda binaria
        SI encontrado ENTONCES
            IMPRIMIR nota y estado
        SINO
            IMPRIMIR "Curso no registrado"
        FIN SI
    FIN FUNCION

    FUNCION simular_cola_revision
        MIENTRAS True HACER
            IMPRIMIR submenú de cola
            LEER opción
            limpiar_terminal()
            SEGÚN opción HACER
                1: AGREGAR solicitud a cola
                2: ATENDER primera solicitud, actualizar nota, registrar historial
                3: MOSTRAR solicitudes
                4: SALIR submenú
            FIN SEGÚN
        FIN MIENTRAS
    FIN FUNCION

    FUNCION mostrar_historial
        SI historial_cambios no está vacío ENTONCES
            IMPRIMIR historial (más reciente primero)
        SINO
            IMPRIMIR "No hay cambios"
        FIN SI
    FIN FUNCION

    FUNCION ejecutar_opcion(opcion)
        SEGÚN opcion HACER
            1: registrar_curso_lista()
            2: mostrar_cursos_lista()
            3: promedio()
            4: cursos_aprobados_reprobados()
            5: buscar_curso()
            6: actualizar_nota()
            7: borrar_curso_lista()
            8: ordenar_por_nota_burbuja()
            9: ordenar_por_nombre()
            10: buscar_curso_binario()
            11: simular_cola_revision()
            12: mostrar_historial()
            13: IMPRIMIR "Saliendo" → RETORNAR FALSO
            OTRO: IMPRIMIR "Opción inválida"
        FIN SEGÚN
        RETORNAR VERDADERO
    FIN FUNCION

// ==================== BUCLE PRINCIPAL ====================
MIENTRAS True HACER
    mostrar_menu()
    LEER opcion
    limpiar_terminal()
    SI ejecutar_opcion(opcion) = FALSO ENTONCES
        SALIR BUCLE
    FIN SI
    LEER respuesta "¿Desea realizar otra operación? (s/n)"
    SI respuesta ≠ "s" ENTONCES
        IMPRIMIR "Programa terminado"
        SALIR BUCLE
    SINO
        limpiar_terminal()
    FIN SI
FIN MIENTRAS

FIN
